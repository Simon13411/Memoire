	SELECT box (et toutes ses infos) en fonction d'un ordre, sous-ordre, genus, subgenus, ...

SELECT B.id_box, B.location, B.speciesrange_id, B.genusrange_id, B.museum, B.paratypes, B.types, Spec.species_range_begin, Spec.species_range_end, Gen.genus_range_begin, Gen.genus_range_end
FROM "Box" B, (
    SELECT P.box_id
    FROM "PopuBox" P, Population P2
    WHERE P.population_id==P2.id_population AND P2.order_id=1 AND P2.suborder_id=1 AND P2.family_id=1 AND P2.subFamily_id=1 AND P2.tribu_id=1 AND P2.genus_id=6 AND P2.subGenus_id=1 AND P2.species_id=2 AND P2.subSpecies_id=1) AS R,
    (SELECT id_speciesrange, range_begin AS species_range_begin, range_end AS species_range_end
    FROM "SpeciesRange" SR) AS Spec,
    (SELECT id_genusrange, range_begin AS genus_range_begin, range_end AS genus_range_end
    FROM "GenusRange" SR) AS Gen
WHERE B.id_box == R.box_id AND Spec.id_speciesrange==B.speciesrange_id AND Gen.id_genusrange==B.genusrange_id

WHERE I.box_id == R.box_id
